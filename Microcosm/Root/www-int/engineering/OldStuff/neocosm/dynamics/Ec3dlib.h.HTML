struct ecVector {
  float x, y, z;
};

struct ecVertex {
  float x, y, z;
  float u, v;
};

// As per VRML 2 spec, rotation assumes vector component is normalized
// and r is in radians
struct ecRotation {
  float x,y,z,r;
};

struct ecMatrix {
  float fElements[12]; 
};

struct ecMaterial {
  float  fAmbient, fDiffuse, fSpecular;
  float  fTransparency;
  BYTE   r, g, b;
  char  *strTexture;
};

struct ecPolygon {
  int  MaterialIndex;
  int  NumVertices;
  int  *Vertices;
};

struct ecMesh {
  int         NumPolygons, NumVertices, NumMaterials;
  int         PolySize, VerticesSize, MaterialsSize;
  BOOL        IsDirty;     
  ecV3d       LocalCenter;           // Center of object, used for bounding 
                                     // sphere calculation.
  float       Radius;                // radius, used for bounding sphere
  ecVertex   *Vertices;              // array of vertices, used for building polygons
  ecPolygon  *Polygons;              // array of polygons
  ecMaterial *Materials;             // array of materials, used for building polygons

  ecMatrix   *StaticMtx;             // matrix that indicates the static (unanimated) geometry
  ecMatrix   *Matrix;                // matrix that transforms to parent's 
                                     // coordinate system (or to world, if this is root).
  char       *Name;                  // Drawn from DEF name, stripped of suffix
  int         Tec;                   // [??? tony] tec, currently not assumed to be unique. 
  ecMesh     *Parent;
  ecMesh     *NextSibling;
  ecMesh     *FirstChild;

  // AC specific members
  // XXX - should be able to eliminate this DrawingType
  char           DrawingType; 
  ACPolySetData *PolySetData;
  long           PolySet;

  // Needed for cartoons:
  long           CartoonStyle;
  ACContext      Context;
  CartoonData   *CartoonData;
};

// [??? tony] These animation structures could use some work
// I'd suggest investigating parsers, designing our anim feature 
// list before reworking these...
struct ecAnimation {
  int               Index;           // Identifies uniquely the animation within 
                                     // the avatars array of animations.
  int               NumFrames;       // Total number of frames of this animation
  int               LoopCount;       // Number of times loop will be repeated
  UInt32            Duration;        // Length of animation duration in milliseconds.
  ecKeyAnimation   *KeyAnimation;    
  UInt32           *nTimePerFrame;   // Time in milliseconds.  Identical to 
                                     // the keys in VRML, only in units of ms.
  // transient
  Int64             StartTime;       // Start time in milliseconds?
  BOOL              ForceLoop;       // Animation is currently looping
  BOOL              PlayWithReset;

  int               CurrentLoop;     // Current loop being executed
  UInt32            TotalDuration;   // The total length of time in milliseconds
                                     // that animation is expected to be looping.
}                                    // (LoopCount*Duration)

struct ecKeyAnimation {
  int          NumAnimatedParts;     // The number of body parts to be animated.
  BOOL         Interpolate;          // TRUE if animation will interpolate between
                                     // frames (XXX - this is not supported yet as 
                                     // of 2/10/97).
  ecKeyData   *KeyData;              // An array of structures, w/ one element
                                     // for each body part, that contains the info
                                     // necessary to animate that body part.
};

struct ecKeyData {
  ecMatrix **pTransformations;       // An array of matrices, with one matrix
                                     // for each frame in the animation
  ecMesh    *Mesh;             
};




// Any Channel Specific structures

struct EcSlots {
  int          NumRegPoints;
  char       **sRegPointNames;       // An array of strings containing all the
                                     // names of the registration points.
  ecMesh     **pRpBodyParts;         // Array of pointers to Polytree body parts,
                                     // corresponding to each registration point.
                                     // These must be placed in the same order
                                     // as the sRegPointNames.
  ecVector    *vRpOffsets;           // Offset for each registration pt, 
                                     //        in same order as above.
  ecVector    *vRpOrientation;       // Vector signifying orientation of slot.
  float       *fLastYaw;             // The last yaw computed from atan2.
};

struct SPick
{
  EPickType    eType;
  int          nCellX, nCellY;
  void        *pObject;
  RwPolygon3d *pPolygon;
};



// [??? tony] This has always been ugly.
// Let's lose it, please 
struct ecAvatar
{
  int           NumAnimations;
  int           NumRunning;          // NumRunning is the number of
                                     // entries in the nCurrentRunning array
                                     // that are relevant.
  int           NumAllocated;        // nNumAllocated gives the size for both
                                     // the Animation array and the
                                     // the nCurrentRunning.
  int           NumMacros;           // The macro variables parallel the 
  int           NumRunningMacros;    // animation variables above.
  int           NumMacrosAllocated;    

  int           NumLods;
  int           CurrentLod;
  BOOL          FlippingYZ;
  ecMesh      **pLodArray;           // an array of pointers to polytrees, each
                                     // one corresponding to one LOD element.
  ecAnimation **pAnimation;          // an array of pointers to animation
                                     // structures (objects).
  int          *nCurrentRunning;     // Array of indices to the animations that
                                     // are currently running.
  ecAnimMacro **pAnimMacro;          // an array of pointers to animation
                                     // structures (objects).
  int          *nRunningMacros;      // Array of indices to the animations that
                                     // are currently running.
  void         *UserData;            // This is holding data for the slots in
                                     // EC, but is made void *, so as not to
                                     // be EC specific.
};
